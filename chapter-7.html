# 讀取檔案

sheet_name_1 <- "Lux_Hotel" #這裡輸入您要讀取的sheet的名稱

# 使用read.xlsx函數讀取指定的工作表
lux.hotel.df <- read.xlsx(file_path, sheet = sheet_name_1)
